#include <stdio.h>
#include <time.h>

int main() {
    int num;
    int i, key, element_found = 0;
    printf("Enter number of elements: ");
    scanf("%d", &num);
    int arr[num];
    printf("\nEnter the elements: ");
    for (i = 0; i < num; i++) {
        scanf("%d", &arr[i]);
    }

    printf("\nEnter the element to be searched: ");
    scanf("%d", &key);

    clock_t start, end;
    double cpu_time_used;
    start = clock();

    for (i = 0; i < num; i++) {
        if (key == arr[i]) {
            element_found = 1;
            break;
        }
    }

    end = clock();
    cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;

    if (element_found == 1)
        printf("Element found at index %d\n", i + 1);
    else
        printf("Element not found\n");

    printf("Runtime: %f seconds\n", cpu_time_used);
    return 0;
}
