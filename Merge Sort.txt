#include<stdio.h>
void merge(int a[],int low,int mid,int high){
    int temp[high-low+1],i=0;
    int left=low,right=mid+1;
    while(left<=mid && right<=high){
        if(a[left]<=a[right]){
            temp[i]=a[left];
            left++;
        }
        else{
            temp[i]=a[right];
            right++;
        }
        i++;
    }
    while(left<=mid){
        temp[i]=a[left];
        left++;
        i++;
    }
    while(right<=high){
        temp[i]=a[right];
        right++;
        i++;
    }
    for(i=low;i<=high;i++){
        a[i]=temp[i-low];
    }
}
int sort(int a[],int low,int high){
    if(low<high){
        int mid=(high+low)/2;
        sort(a,low,mid);
        sort(a,mid+1,high);
        merge(a,low,mid,high);
    }
}
int main(){
    int n;
    printf("enter number of elements:");
    scanf("%d",&n);
    int a[n];
    printf("enter the elements:");
    for(int i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    sort(a,0,n-1);
    for(int i=0;i<n;i++){
        printf("%d ",a[i]);
    }
}
